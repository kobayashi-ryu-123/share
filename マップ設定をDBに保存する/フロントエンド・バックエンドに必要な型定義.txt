// settings.model.ts
// カテゴリ定義
export type Category = 'age' | 'time' | 'target'
// サマリ設定
export type SummarySetting = {
  category: Category
  display: boolean
  order: number
}
// マーカーフィルタ設定（項目）
export type FilterItem = {
  key: string
  checked: boolean
}
// マーカーフィルタ設定
export type FilterSetting = {
  category: Category
  items: FilterItem[]
}
// 設定
export type Settings = {
  summary: SummarySetting[]
  filters: FilterSetting[]
}

■ViewModel変換
const summaryView = settings.summary.map(s => ({
  ...s,
  label: CATEGORY_LABELS[s.category]
}))

const filterView = settings.filters.map(f => ({
  category: f.category,
  label: CATEGORY_LABELS[f.category],
  items: f.items.map(i => ({
    ...i,
    label: ITEM_LABELS[f.category][i.key]
  }))
}))

■絞り込みロジック
function filterAccidents(accidents, settings) {
  const filterMap = {}

  settings.filters.forEach(f => {
    const checked = f.items.filter(i => i.checked).map(i => i.key)
    if (checked.length) filterMap[f.category] = checked
  })

  return accidents.filter(acc =>
    Object.entries(filterMap).every(([cat, vals]) =>
      vals.includes(acc[cat])
    )
  )
}
